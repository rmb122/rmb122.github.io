```cpp
#include <iostream>
#include <string>
#include <cstdlib>

using namespace std;

void hanoi(int n, string from,string to,string buffer)
{
	if (n == 1)
	{
		cout << "Move a plate from " + from + " to " + to << endl;
	}
	else
	{
		hanoi(n - 1, from, buffer, to);
		hanoi(1, from, to, buffer);
		hanoi(n - 1, buffer, to, from);
	}
}

int main()
{
	int n;
	cin >> n;
	hanoi(n, "A", "C", "B");
	system("pause");
    return 0;
}
```
